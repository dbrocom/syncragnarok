prt_in,243,168,4	script	Jovovich	91,{
	mes "[Hairdresser Jovovich]";
	mes "¡Bienvenido!";
	mes "¿Cómo puedo ayudarte?";
	next;
	mes "[Hairdresser Jovovich]";
	if (sex) mes "¡Oh, no! Tu cabello está dañado... Parece que necesitas tratamiento profesional. Ven siéntate aquí...";
	else {
		if (rand(20) > 11) {
			mes "¡Dios! ¡Por favor! Que dañado que está tú cabello... No es bueno que esto quede así...";
			next;
			mes "[Hairdresser Jovovich]";
			mes "¿Me dejarás arreglar tu cabello? Por favor...";
		}
		else mes "¡Wow! Tu cabello quedará perfecto una vez teñido... ¿Te parece teñir el pelo para empezar un cambio?";
	}
	next;
	while(1) {
		switch(select("Dye Hair:Tips and Information:Cancel")) {
		case 1:
			mes "[Hairdresser Jovovich]";
			mes "Si, si... Buena elección...";
			mes "Bien... Esta es tú oportunidad de tener un nuevo look... Elige que color quieres...";
			next;
			while(1) {
				if (.@choose_success == 1) {
					mes "[Hairdresser Jovovich]";
					mes "¿Qué piensas? ¿Te gustaría un color nuevo?";
					next;
					if (select("Yes:No") == 1) {
						mes "[Hairdresser Jovovich]";
						mes "Ok... Elige que color quieres...";
						next;
					}
					else  {
						mes "[Hairdresser Jovovich]";
						mes "Hmm, estoy algo desilusionado... Quise hacer un mejor trabajo... Te prometo que la próxima será mejor... Por favor vuelve...";
						close;
					}
				}

				while(1) {
					switch(select("Red, please.:Yellow, please.:Violet, please.:Orange, please.:Green, please.:Blue, please.:White, please.:Black, please.:Actually, I like my hair as it is.")) {
					case 1: set .@headpalette,8; break;
					case 2: set .@headpalette,1; break;
					case 3: set .@headpalette,2; break;
					case 4: set .@headpalette,3; break;
					case 5: set .@headpalette,4; break;
					case 6: set .@headpalette,5; break;
					case 7: set .@headpalette,6; break;
					case 8: set .@headpalette,7; break;
					case 9:
						if (.@choose_success != 0) {
							mes "[Hairdresser Jovovich]";
							mes "Te debe gustar tu color...";
							close;
						}
						mes "[Hairdresser Jovovich]";
						mes "¿Qué? ¿No vas a teñir tu cabello? Que triste estoy...";
						close;
					}

					if (.@headpalette == getlook(VAR_HEADPALETTE)) {
						mes "[Hairdresser Jovovich]";
						mes "Pero ese es el color que ya tienes... Por favor selecciona otro...";
						next;
					}
					else {
						switch(.@headpalette) {
						case 1: callsub S_NoDye,976,"yellow"; break;
						case 2: callsub S_NoDye,978,"violet"; break;
						case 3: callsub S_NoDye,979,"orange"; break;
						case 4: callsub S_NoDye,980,"green"; break;
						case 5: callsub S_NoDye,981,"blue"; break;
						case 6: callsub S_NoDye,982,"white"; break;
						case 7: callsub S_NoDye,983,"black"; break;
						Case 8: callsub S_NoDye,975,"red"; break;
						}
						if (zeny < 1000) {
							mes "[Hairdresser Jovovich]";
							mes "El costo es de 1000 zeny. ¿Te alcanza?";
							close;
						}

						switch(.@headpalette) {
						case 1: delitem 976,1; break; //Lemon_Dyestuffs
						case 2: delitem 978,1; break; //Violet_Dyestuffs
						case 3: delitem 979,1; break; //Orange_Dyestuffs
						case 4: delitem 980,1; break; //Darkgreen_Dyestuffs
						case 5: delitem 981,1; break; //Cobaltblue_Dyestuffs
						case 6: delitem 982,1; break; //White_Dyestuffs
						case 7: delitem 983,1; break; //Black_Dyestuffs
						case 8: delitem 975,1; break; //Scarlet_Dyestuffs
						}
						set Zeny,Zeny-1000;
						setlook VAR_HEADPALETTE,.@headpalette;
						set .@choose_success,1;
						break;
					}

				}

			}

		case 2:
			mes "[Hairdresser Jovovich]";
			mes "Cuando te sientas decaído, estés triste, cuando quieras impresionar a alguien, o simplemente por un cambio...";
			next;
			mes "[Hairdresser Jovovich]";
			mes "Para ese momento especial, ¿no te gustaría un estilo de cabello bien personal? Siempre que tengas la tintura apropiada, haré que tu cabello luzca maravilloso.";
			next;
			mes "[Hairdresser Jovovich]";
			mes "Y no te preocupes por el costo. La belleza que obtendrás es como obtener todo...";
			next;
			mes "[Hairdresser Jovovich]";
			mes "* Tasas e Información *";
			mes "- 1 Dyestuffs ítem de el color de cabello que quieres.";
			mes "- 1000 zeny de costo.";
			break;

		case 3:
			mes "[Hairdresser Jovovich]";
			mes "Hombre o Mujer...";
			mes "Todos tienen el derecho y la obligación de verse hermosos.";
			close;
		}

	}

S_NoDye:
	if (countitem(getarg(0)) == 0) {
		mes "[Hairdresser Jovovich]";
		mes "¡¿Eh?! Pero necesitas el ítem '"+getitemname(getarg(0))+"' para teñir tu cabello."+getarg(1)+"...";
		close;
	}
	return;
}
